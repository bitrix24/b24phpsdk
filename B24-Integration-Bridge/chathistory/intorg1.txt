function transformDealData($sourceData, $mapping) {
    $result = ['fields' => []];
    
    foreach ($mapping as $sourceField => $targetField) {
        if (isset($sourceData[$sourceField])) {
            $result['fields'][$targetField] = $sourceData[$sourceField];
        }
    }
    
    return $result;
}

function sendToESB($dealData) {
    $url = ESB_WEBHOOK_URL . 'crm.deal.add.json';

function callBitrix24($url, $params) {
    $curl = curl_init();
    curl_setopt_array($curl, [
        CURLOPT_URL => $url,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_POST => true,
        CURLOPT_POSTFIELDS => http_build_query($params),
        CURLOPT_HTTPHEADER => ['Content-Type: application/x-www-form-urlencoded']
    ]);
    
    $response = curl_exec($curl);
    curl_close($curl);
    
    return json_decode($response, true);
}

}
